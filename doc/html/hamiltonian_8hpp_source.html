171  virtual void generate(T &hmat, T&smat, const AtomicStruct &bi, const AtomicStruct &bj){
172  // Just for easy reference
173  int nai = bi.NumOfAtoms();
174  int naj = bj.NumOfAtoms();
175  int noi = bi.NumOfOrbitals();
176  int noj = bj.NumOfOrbitals();
177 
178  // Most of the matrix elements are zeros. So, we'll only change the
179  // non zero elements below.
180  hmat = zeros<T>(noi, noj);
181  smat = zeros<T>(noi, noj);
182 
183  // Lets find the neighbors.
184  int io = 0;
185  for(int ia = 0; ia != nai; ++ia){
186  AtomicStruct atomi = bi(ia); // extract atom ia
187  int ni = atomi.NumOfOrbitals(); // number of orbitals in atom ia
188 
189  int jo = 0;
190  for(int ja = 0; ja != naj; ++ja){
191  AtomicStruct atomj = bj(ja); // extract atom ja
192  int nj = atomj.NumOfOrbitals(); // number of orbitals in atom ja
193  // generate Hamiltonian matrix between orbitals of
194  // atom i and atom j
195  hmat(span(io,io+ni-1), span(jo,jo+nj-1)) = genTwoAtomHam(atomi, atomj);
196  smat(span(io,io+ni-1), span(jo,jo+nj-1)) = genTwoAtomOvl(atomi, atomj);
197  jo += nj;
198  }
199 
200  io += ni;
201  }
202  }
